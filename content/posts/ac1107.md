N = int(input())
M = int(input())
enable = {str(x) for x in range(10)}
if (M != 0):
    enable -= set(input().split())

result = abs(N - 100)
for i in range(1000001):
    is_true = True
    for c in str(i):
        if (c not in enable):
            is_true = False
            break
    if is_true is True:
        result = min(result, abs(N - i) + len(str(i)))

print(result)
