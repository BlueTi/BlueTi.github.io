---

- 주유소를 거리가 짧은 순으로 정렬하여 현재 연료로 접근할 수 있는 주유소로 접근한다.
- 넣을 수 있는 기름의 양이 큰 순서로 접근한 주유소 목록 정렬하여 큰 양을 넣는다.

---



```java
import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.Comparator;
import java.util.PriorityQueue;
import java.util.StringTokenizer;

//처음에 comparable을 이용하여 정렬하려고 했으나 두가지가 필요하여 클래스에 주유소 정보만 담는다.
class Node {
    int dist, fuel;

    public Node(int dist, int fuel) {
        this.dist = dist;
        this.fuel = fuel;
    }
}

public class Main {
    static BufferedReader br;
    static BufferedWriter bw;
    static StringBuilder sb;

    public static void main(String[] args) throws IOException {
        br = new BufferedReader(new InputStreamReader(System.in));
        bw = new BufferedWriter(new OutputStreamWriter(System.out));
        int N = Integer.parseInt(br.readLine());
        //정렬을 새로 정의하여 거리가 가까운 순서로 주유소를 입력받아 정렬한다.
        PriorityQueue<Node> stations = new PriorityQueue<Node>(new Comparator<Node>() {
            @Override
            public int compare(Node o1, Node o2) {
                return o1.dist - o2.dist;
            }
        });
        //앞으로 들어올 방문한 주유소 목록을 넣을 수 있는 가스가 큰 순서로 정렬한다.
        PriorityQueue<Node> gas = new PriorityQueue<Node>(new Comparator<Node>() {
            @Override
            public int compare(Node o1, Node o2) {
                return o2.fuel - o1.fuel;
            }
        });
        StringTokenizer st;
        for (int i = 0; i < N; i++) {
            st = new StringTokenizer(br.readLine());
            stations.add(new Node(Integer.parseInt(st.nextToken()), Integer.parseInt(st.nextToken())));
        }
        st = new StringTokenizer(br.readLine());
        int L = Integer.parseInt(st.nextToken()), P = Integer.parseInt(st.nextToken());

        int count = 0;
        //P가 현재 연료이므로 연료량이 L이 넘거나 같을때까지 반복한다.
        while (P < L) {
            //현재 연료로 접근 가능한 주유소 목록에 주유소를 추가한다.
            while (!stations.isEmpty() && stations.peek().dist <= P) {
                Node n = stations.poll();
                gas.add(n);
            }
            //연료(P)로 목표점(L)까지 가지 못했는데 접근가능한 주유소가 없을 경우 도착할 수 없으므로 -1을 출력한다.
            if (gas.isEmpty()) {
                count = -1;
                break;
            }
            //현재연료량에 접근가능한 주유소 중 가장 많은 양의 기름을 넣어준다.
            P += gas.poll().fuel;
            count++;
        }
        bw.write(count + "");
        bw.flush();
    }
}
```
